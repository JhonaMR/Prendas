/**
 * üöÄ SERVIDOR PRINCIPAL - POSTGRESQL

 * Este es el punto de entrada del backend
 * Configura Express, middlewares, rutas, inicializa PostgreSQL y arranca el servidor
 */
 
require('dotenv').config();
const express = require('express');
const cors = require('cors');
const path = require('path');
const fs = require('fs');
const https = require('https');
 
// Importar configuraci√≥n y rutas
const { initDatabase, closePool } = require('./config/database');
const apiRoutes = require('./routes');
const logger = require('./utils/logger');
const configurationManager = require('./config/configurationManager');
const postgres = require('./config/postgres');
const { trackRemoteClient, logDatabaseOperation } = require('./middleware/remoteClientTracking');
 
// Crear aplicaci√≥n Express
const app = express();
 
// Obtener puerto y host de variables de entorno
const PORT = process.env.PORT || 3000;
const HOST = process.env.HOST || '0.0.0.0';
const USE_HTTPS = process.env.USE_HTTPS !== 'false'; // true por defecto
 
// ==================== MIDDLEWARES ====================
 
/**
 * CORS - Permitir peticiones desde el frontend
 * En desarrollo: http://localhost:5173 (Vite)
 * En producci√≥n: http://IP_SERVIDOR:3000
 */
const corsOptions = {
    origin: process.env.CORS_ORIGIN?.split(',') || ['http://localhost:5173', 'http://localhost:3000'],
    credentials: true,
    optionsSuccessStatus: 200
};
app.use(cors(corsOptions));
 
/**
 * Parser de JSON
 * Permite recibir datos JSON en las peticiones
 */
app.use(express.json());
 
/**
 * Parser de URL-encoded
 * Permite recibir datos de formularios
 */
app.use(express.urlencoded({ extended: true }));
 
/**
 * Logger simple
 * Muestra en consola cada petici√≥n que llega
 */
app.use((req, res, next) => {
    const timestamp = new Date().toISOString();
    logger.info(`${req.method} ${req.path}`);
    next();
});
 
/**
 * Middleware para rastrear conexiones de clientes remotos
 * Registra qu√© IP se usa para cada conexi√≥n a la base de datos
 */
app.use(trackRemoteClient);
app.use(logDatabaseOperation);
 
/**
 * Middleware para verificar que la base de datos est√© lista
 * Retorna 503 Service Unavailable si la BD no est√° disponible
 * Excepto para el endpoint /health que siempre debe estar disponible
 */
app.use((req, res, next) => {
    // Permitir acceso al endpoint de health sin verificaci√≥n
    if (req.path === '/health') {
        return next();
    }
 
    // Verificar estado de conexi√≥n
    const connectionStatus = postgres.getConnectionStatus();
    if (!connectionStatus.connected) {
        logger.warn(`‚ö†Ô∏è Solicitud rechazada: Base de datos no disponible (${req.method} ${req.path})`);
        return res.status(503).json({
            success: false,
            message: 'Base de datos no disponible. Por favor, intente m√°s tarde.',
            status: 'service_unavailable'
        });
    }
 
    next();
});
 
// ==================== RUTAS ====================
 
/**
 * Rutas del API
 * Todas las rutas empiezan con /api
 */
app.use('/api', apiRoutes);
 
/**
 * Servir archivos est√°ticos del frontend (cuando est√© compilado)

/**
 * Iniciar servidor en el puerto configurado
 */
async function startServer() {
    try {
        // Inicializar configuraci√≥n con detecci√≥n de red
        logger.info('üîß Inicializando configuraci√≥n con detecci√≥n de red...');
        await configurationManager.initializeConfiguration();
        logger.info('‚úÖ Configuraci√≥n inicializada correctamente');
 
        // Inicializar pool de conexiones con fallback
        logger.info('üóÑÔ∏è Inicializando pool de conexiones con fallback...');
        await postgres.initPoolWithFallback();
        logger.info('‚úÖ Pool de conexiones inicializado correctamente');
 
        // Validar conectividad de base de datos
        logger.info('üè• Validando conectividad de base de datos...');
        const isHealthy = await postgres.healthCheck();
        if (!isHealthy) {
            logger.warn('‚ö†Ô∏è Health check fall√≥, pero continuando con la inicializaci√≥n...');
        } else {
            logger.info('‚úÖ Base de datos lista para aceptar solicitudes');
        }
 
        // Crear servidor HTTPS o HTTP seg√∫n configuraci√≥n
        let server;
        const protocol = USE_HTTPS ? 'https' : 'http';
        
        if (USE_HTTPS) {
            const certDir = path.join(__dirname, '../certs');
            const keyPath = path.join(certDir, 'server.key');
            const certPath = path.join(certDir, 'server.crt');
            
            // Verificar si existen los certificados
            if (!fs.existsSync(keyPath) || !fs.existsSync(certPath)) {
                console.error('\n‚ùå Certificados SSL no encontrados');
                console.error(`   Ejecuta: node backend/scripts/generate-ssl-cert.js`);
                process.exit(1);
            }
            
            const options = {
                key: fs.readFileSync(keyPath),
                cert: fs.readFileSync(certPath)
            };
            
            server = https.createServer(options, app);
        } else {
            const http = require('http');
            server = http.createServer(app);
        }
 
        // Iniciar servidor
        server.listen(PORT, HOST, () => {
            const config = configurationManager.getConfiguration();
            console.log('\n' + '='.repeat(60));
            console.log('üöÄ SERVIDOR BACKEND INICIADO');
            console.log('='.repeat(60));
            console.log(`üìç URL Local:    ${protocol}://localhost:${PORT}`);
            console.log(`üìç URL Red:      ${protocol}://${getLocalIP()}:${PORT}`);
            console.log(`üìÅ Entorno:      ${config.NODE_ENV}`);
            console.log(`üåê IP Detectada: ${config.DETECTED_IP || 'No detectada'}`);
            console.log(`üóÑÔ∏è  Base de datos: PostgreSQL (${config.DB_HOST}:${config.DB_PORT})`);
            console.log(`üîê CORS habilitado para:`, corsOptions.origin.join(', '));
            console.log(`üîí Protocolo:    ${protocol.toUpperCase()}`);
            console.log('='.repeat(60));
            console.log('\n‚úÖ El backend est√° listo para recibir peticiones');
            console.log('üìù Los logs de peticiones aparecer√°n abajo:\n');
        });
    } catch (error) {
        logger.error('‚ùå Error al iniciar servidor:', error);
        process.exit(1);
    }
}
 
/**
 * Obtener IP local de la m√°quina
 * √ötil para acceder desde otros PCs de la red
 */
function getLocalIP() {
    const { networkInterfaces } = require('os');
    const nets = networkInterfaces();
    
    for (const name of Object.keys(nets)) {
        for (const net of nets[name]) {
            // IPv4 y no loopback
            if (net.family === 'IPv4' && !net.internal) {
                return net.address;
            }
        }
    }
    
    return 'localhost';
}
